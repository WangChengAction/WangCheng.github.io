背景

这次的实战其实很简单，就是在之前小猪短租爬虫的基础上，将数据存储至MongoDB数据库中，然后在屏幕上显示出前三页中日租金大于500的房源信息。

代码

具体代码如下：

from bs4 import BeautifulSoup
import requests
import time
import pymongo


client = pymongo.MongoClient('localhost',27017)
xiaozhu = client['xiaozhu']
sheet_info = xiaozhu['sheet_info']

urls = ["http://bj.xiaozhu.com/search-duanzufang-p{}-0/".format(str(i)) for i in range(1,4)]

def collect_info(url,data=None):
    response = requests.get(url)
    soup = BeautifulSoup(response.text,'lxml')
    time.sleep(2)

    #标题，地址，日租金，第一张图片链接，房东图片链接，房东性别，房东名字
    titles = soup.select('body > div.wrap.clearfix.con_bg > div.con_l > div.pho_info > h4 > em')
    addresses = soup.select('body > div.wrap.clearfix.con_bg > div.con_l > div.pho_info > p > span.pr5')
    day_prices = soup.select('div.day_l')
    images = soup.select('div.pho_show_l > div > div:nth-of-type(2) > img')
    host_imgs = soup.select('div.js_box.clearfix > div.member_pic > a > img')
    host_sexes = soup.select('div.js_box.clearfix > div.w_240 > h6 > span')
    host_names = soup.select('div.js_box.clearfix > div.w_240 > h6 > a')



    for title,address,day_price,image,host_img,host_sex,host_name in zip(titles,addresses,day_prices,images,host_imgs,host_sexes,host_names):

        #判断性别
        sex = ""
        for s in host_sex.get('class'):

            if s == 'member_girl_ico':
                sex = "女士"
            elif s == 'member_boy_ico':
                sex = "先生"
            else:
                sex = "未知"

        price = list(day_price.stripped_strings)




        data = {
            '标题' : title.get_text(),
            '地址' : address.get_text().strip(),#地址去除多于的空格和回车
            '日租金' : int(price[1]),
            '图片地址' : image.get('src'),
            '房东头像' : host_img.get('src'),
            '房东性别' : sex,
            '房东姓名' : host_name.get('title')
        }



        sheet_info.insert_one(data)

def find_link(url,data=None):

    response = requests.get(url)
    soup = BeautifulSoup(response.text,'lxml')

    link = soup.select('ul > li > a[class="resule_img_a"]')

    for l in link:
        url_pages = l.get('href')
        collect_info(url_pages)


def find_in_db():
    for item in sheet_info.find():
        if (item['日租金'] >= 500):
            print(item)

for url in urls:
    find_link(url)

find_in_db()
