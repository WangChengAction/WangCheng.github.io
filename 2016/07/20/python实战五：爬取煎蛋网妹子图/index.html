研究情况

爬煎蛋网真是爬的我外焦内嫩，第一次按照之前爬别的网站的模式进行爬取，大概爬了60来张图片就被封IP了，于是开始学习构建IP代理池，经过一系列的鼓捣（感谢麻瓜编程），加上了IP代理池，然后继续爬，爬了大概100多张后又给我禁了，然后在知乎上搜了搜，正好发现一篇描述煎蛋网反爬机制文章,发现可能是UA的问题，通过 User-Agent 判定是不是同一个用户访问的,于是研究了下答主的代码，加入了一个UA池，现在暂时是一直爬着没出现被封的问题，只是发现会出现图片重复下载的现象，我想着把待下载的链接放在一个set内，但还是没有用。

urls= ["http://i.jandan.net/ooxx/page-{}#comments".format(str(i)) for i in range(2000,2100)]


#代理池
resp = requests.get("http://tor1024.com/static/proxy_pool.txt")
ips_txt = resp.text.strip().split("\n")
ips = []
for i in ips_txt:
    try:
            k = json.loads(i)
            ips.append(k)
    except Exception as e:
            print(e)

#下载图片
def collect_img(url):

    response = requests.get(url,headers=headers,proxies=random.choice(ips),timeout=6)
    soup = BeautifulSoup(response.text,'lxml')

    # time.sleep(4)

    path_img = "/Users/wangcheng/Desktop/JianDan/"
    # images = soup.select('div > div > div.text > p > img')
    images = soup.select('div > div > div.text > p > a[target="_blank"]')
    download_list = set()
    for img in images:
        image_url = img.get('href')
        download_list.add(image_url)

    for item in download_list:
        urllib.request.urlretrieve(item,path_img+item[-10:])
        time.sleep(2)
        print(item)


for url in urls:
    collect_img(url)
